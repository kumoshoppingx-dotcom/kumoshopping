<script>
  // THE ENGINE: Automatically sorts products into the right section
  async function fetchProducts() {
    try {
      const response = await fetch('products.json');
      const data = await response.json();
      
      data.forEach(item => {
        // Find the specific grid based on category (e.g., id="grid-tech")
        const targetGrid = document.getElementById(`grid-${item.category}`);
        
        if (targetGrid) {
          const cardHtml = `
            <div class="card">
              <div>
                <span class="badge ${item.badgeClass}">${item.badge}</span>
                <span class="platform-tag" style="color:var(--${item.platformClass})">${item.platform}</span>
              </div>
              <img src="${item.img}" alt="${item.name}">
              <h3>${item.name}</h3>
              <p style="font-size:13px; color:#888;">${item.desc}</p>
              <div class="price">${item.price}</div>
              <a href="${item.link}" class="btn ${item.platformClass}" target="_blank">Buy on ${item.platform}</a>
            </div>
          `;
          targetGrid.innerHTML += cardHtml;
        }
      });
    } catch (err) {
      console.error("Error loading products. Make sure products.json exists!", err);
    }
  }

  fetchProducts();
</script>
